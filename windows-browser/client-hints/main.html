Windows Browser: Client Hint Header Config

This page opts in to all known Client Hint request headers with Accept-CH and displays what the Windows browser sends for the top-level page and for iframe fetches.

Remote config examples

Use these targeted examples against privacy-test-pages.site or *.third-party.site to verify different outcomes.

{
  "clientBrandHint": {
    "state": "enabled",
    "settings": {
      "domains": [
        { "domain": "privacy-test-pages.site", "brand": "DuckDuckGo" },
        { "domain": "*.third-party.site", "masqueradeAsChrome": true },
        {
          "domain": "privacy-test-pages.site",
          "headers": {
            "sec-ch-ua-platform": "\"Windows\"",
            "sec-ch-ua-platform-version": "\"15.0.0\"",
            "sec-ch-ua-full-version": "\"109.0.0.0\""
          }
        }
      ]
    }
  },
  "customUserAgent": {
    "state": "enabled",
    "features": {
      "userAgentStrategies": {
        "state": "enabled",
        "settings": {
          "strategies": [
            { "domain": "*.third-party.site", "strategy": "ChromeUA" }
          ]
        }
      }
    }
  }
}
